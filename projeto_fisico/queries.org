   Quantos cinemas terminam o numero de telefone com o mesmo digito?
  #+begin_src sql
    CREATE FUNCTION endPhoneDig(endnum integer)
           RETURNS varchar(14) -- digito de phone final
           LANGUAGE plpgsql
        AS
    $$
    DECLARE
        dig varchar(14);
    BEGIN
        SELECT cont_tele
        INTO dig
               FROM CINEMA AS c
               WHERE c.cont_tele ~ ($1 || '$');

        RETURN dig;
    END;
    $$;
